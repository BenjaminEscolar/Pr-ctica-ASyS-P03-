grid on;

Find the energy of the signal depicted in Fig.P3.1-1c

% Fig.P3.1-1c

% Por la naturaleza de la señal podemos definir este intervalo
n = -3:3;
sympref('HeavisideAtOrigin',1);
% Hacemos anónima a la funcion
xn = @(n) 3*n.*(heaviside(n+3)-heaviside(n-4));

% Revisando el ejercicio 3 la formula para el cálculo general de la energía de una señal
% discreta finita será: E=(sum(f(n).^2) con el intervalo n definido
%Aplicamos la formula para la señal xn, y obtenemos

Exn = sum(xn(n).^2)
